72void Foam::dgFluxSolver::makeONB
73(
74 const vector& n,
75 vector& t1,
76 vector& t2
77) const
78{
79 // Construct t1 not colinear with n
80 vector a = (mag(n.x()) < 0.9) ? vector(1,0,0) : vector(0,1,0);
81 t1 = a - (a & n)*n; // remove normal component
82 scalar m = mag(t1) + SMALL; // normalize with guard
83 t1 /= m;
84 t2 = n ^ t1; // right-handed basis
85}

87void Foam::dgFluxSolver::decomposeU
88(
89 const vector& U,
90 const vector& n,
91 vector& Un,
92 vector& Ut1,
93 vector& Ut2
94) const
95{
96 // Build orthonormal basis (n, t1, t2)
97 vector t1, t2;
98 makeONB(n, t1, t2);
99
100 // Components of U along basis directions
101 const scalar Un_scalar = U & n; // projection on n
102 const scalar Ut1_scalar = U & t1; // projection on t1
103 const scalar Ut2_scalar = U & t2; // projection on t2
104
105 // Convert back to vector form along the basis
106 Un = Un_scalar * n;
107 Ut1 = Ut1_scalar * t1;
108 Ut2 = Ut2_scalar * t2;
109}
