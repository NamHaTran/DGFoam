30Description
31 Enumeration for DG face positions on reference elements.
32
33 The face positions are defined according to the vertex ordering
34 convention used in:
35
36 Sherwin, S.J., Karniadakis, G.E.
37 "Spectral/hp Element Methods for Computational Fluid Dynamics",
38 2nd Edition, Oxford University Press, 2005.
39
40 The enumeration encodes the relative position of a face within a
41 reference element using the same vertex labels (Aâ€“H) as in the
42 literature, ensuring consistent evaluation of DG face integrals.

194inline dgFacePosition mapFacePositionFromTet(const dgFacePositionOnTet f)
195{
196 switch (f)
197 {
198 case dgFacePositionOnTet::ACE: return dgFacePosition::ACEG;
199 case dgFacePositionOnTet::BCE: return dgFacePosition::BDFH;
200 case dgFacePositionOnTet::ABE: return dgFacePosition::ABEF;
201 case dgFacePositionOnTet::ABC: return dgFacePosition::ABCD;
202 case dgFacePositionOnTet::NONE: return dgFacePosition::NONE;
203 default:
204 FatalErrorInFunction << "Invalid dgFacePositionOnTet enum"
205 << abort(FatalError);
206 }
207
208 return dgFacePosition::ABCD;
209}
