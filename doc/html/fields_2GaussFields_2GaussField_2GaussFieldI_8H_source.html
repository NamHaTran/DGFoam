51template<class Type>
52GaussField<Type>::GaussField
53(
54 const dofField<Type>* dofField,
55 label cellID,
56 const dgGeomMesh* mesh
57)
58:
59 dofField_(dofField),
60 cellID_(cellID),
61 mesh_(mesh),
62 cellField_(mesh, &(*dofField)[cellID]),
63 faceField_(cellID, mesh)
64{
65 if (!dofField_ || !mesh_)
66 {
67 FatalErrorInFunction
68 << "Null pointer in GaussField(dofField,cellID,mesh)"
69 << abort(FatalError);
70 }
71
72 // Build neighbor DOF list
73 const labelList& ngh = mesh_->cells()[cellID_]->neighborCells();
74 List<const cellDof<Type>*> cellsDof(ngh.size() + 1);
75
76 cellsDof[0] = &(*dofField_)[cellID_];
77
78 forAll(ngh, i)
79 {
80 const label nid = ngh[i];
81 cellsDof[i + 1] = (nid >= 0 ? &(*dofField_)[nid] : nullptr);
82 }
83
84 faceField_.setCellsDof(cellsDof);
85}
