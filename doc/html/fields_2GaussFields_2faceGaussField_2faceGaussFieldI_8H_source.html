61// From list of DOFs (P-cell + neighbour cells)
62template<class Type>
63faceGaussField<Type>::faceGaussField
64(
65 List<const cellDof<Type>*>& cellsDof,
66 const dgGeomMesh* mesh
67)
68:
69 mesh_(mesh),
70 cellID_(cellsDof[0]->cellID()),
71 cell_(mesh->cells()[cellID_]),
72 facesID_(&(cell_->faces())),
73 nFaces_(facesID_->size()),
74 faces_(nFaces_),
75 cellsDof_(cellsDof),
76 n_(nFaces_),
77 faceAreas_(nFaces_)
78{
79 // Collect pointers to faces
80 for (label f = 0; f < nFaces_; ++f)
81 {
82 faces_[f] = mesh_->faces()[(*facesID_)[f]];
83 n_[f] = faces_[f]->normal();
84 faceAreas_[f] = faces_[f]->area();
85 }
86
87 nGaussPerFace_ = faces_[0]->gaussPointsOwner().size();
88 nGauss_ = nFaces_ * nGaussPerFace_;
89
90 // Get Gauss offset and face's owner patch's ID
91 gaussOffset_.setSize(nFaces_);
92 ownerPatchIDs_.setSize(nFaces_);
93 for (label f = 0; f < nFaces_; ++f)
94 {
95 gaussOffset_[f] = f * nGaussPerFace_;
96
97 if (faces_[f]->isBoundary())
98 {
99 ownerPatchIDs_[f] = faces_[f]->patchID();
100 }
101 else
102 {
103 ownerPatchIDs_[f] = -1;
104 }
105 }
106
107
108 plusValues_.setSize(nGauss_);
109 minusValues_.setSize(nGauss_);
110
111 interpolateFromDof();
112}

242template<class Type>
243void Foam::faceGaussField<Type>::interpolateFromDof()
244{
245 const cellDof<Type>& cellPDof = *cellsDof_[0];
246
247 for (label localFaceI = 0; localFaceI < nFaces_; ++localFaceI)
248 {
249 const label faceID = (*facesID_)[localFaceI];
250 const label owner = mesh_->faceOwner()[faceID];
251 const bool isOwner = (owner == cellID_);
252 bool isBoundaryFace = faces_[localFaceI]->isBoundary();
253
254 const dgGeomFace& face = *faces_[localFaceI];
255
256 // Get Gauss points for owner sides
257 const List<vector>& gpOwner = face.gaussPointsOwner();
258
259 // Get basis functions for owner sides
260 const List<List<scalar>>& ownerBasis = face.ownerBasis();
261
262 const label offset = gaussOffset_[localFaceI];
263
264 if (isOwner)
265 {
266 forAll(gpOwner, i)
267 {
268 minusValues_[offset + i] = pTraits<Type>::zero;
269 for (label k = 0; k < cellPDof.nDof(); ++k)
270 {
271 minusValues_[offset + i] += cellPDof[k] * ownerBasis[i][k];
272 }
273 }
274
275 if (isBoundaryFace)
276 {
277 forAll(gpOwner, i)
278 {
279 plusValues_[offset + i] = minusValues_[offset + i]; //pTraits<Type>::max; //pTraits<Type>::zero;
280 }
281 }
282 else
283 {
284 // Get Gauss points for neighbor sides
285 const List<vector>& gpNeigh = face.gaussPointsNeighbor();
286
287 // Get basis functions for neighbor sides
288 const List<List<scalar>>& neighborBasis = face.neighborBasis();
289
290 const cellDof<Type>& cellNDof = *cellsDof_[localFaceI + 1];
291 forAll(gpNeigh, i)
292 {
293 plusValues_[offset + i] = pTraits<Type>::zero;
294 for (label k = 0; k < cellNDof.nDof(); ++k)
295 {
296 plusValues_[offset + i] += cellNDof[k] * neighborBasis[i][k];
297 }
298 }
299 }
300 }
301 else
302 {
303 // This face has both owner and neighbor sides, so it is internal
304
305 // Get Gauss points for neighbor sides
306 const List<vector>& gpNeigh = face.gaussPointsNeighbor();
307
308 // Get basis functions for neighbor sides
309 const List<List<scalar>>& neighborBasis = face.neighborBasis();
310
311 const cellDof<Type>& cellNDof = *cellsDof_[localFaceI + 1];
312
313 forAll(gpNeigh, i)
314 {
315 minusValues_[offset + i] = pTraits<Type>::zero;
316 for (label k = 0; k < cellPDof.nDof(); ++k)
317 {
318 minusValues_[offset + i] += cellPDof[k] * neighborBasis[i][k];
319 }
320 }
321
322 forAll(gpOwner, i)
323 {
324 plusValues_[offset + i] = pTraits<Type>::zero;
325 for (label k = 0; k < cellNDof.nDof(); ++k)
326 {
327 plusValues_[offset + i] += cellNDof[k] * ownerBasis[i][k];
328 }
329 }
330 }
331 }
332}
