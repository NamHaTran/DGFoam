61// From list of DOFs (P-cell + neighbour cells)
62template<class Type>
63faceGaussField<Type>::faceGaussField
64(
65 List<const cellDof<Type>*>& cellsDof,
66 const dgGeomMesh* mesh
67)
68:
69 mesh_(mesh),
70 cellID_(cellsDof[0]->cellID()),
71 cell_(mesh->cells()[cellID_]),
72 facesID_(&(cell_->faces())),
73 nFaces_(facesID_->size()),
74 faces_(nFaces_),
75 cellsDof_(cellsDof),
76 n_(nFaces_),
77 faceAreas_(nFaces_)
78{
79 // Collect pointers to faces
80 for (label f = 0; f < nFaces_; ++f)
81 {
82 faces_[f] = mesh_->faces()[(*facesID_)[f]];
83 n_[f] = faces_[f]->normal();
84 faceAreas_[f] = faces_[f]->area();
85 }
86
87 nGaussPerFace_ = faces_[0]->gaussPointsOwner().size();
88 nGauss_ = nFaces_ * nGaussPerFace_;
89
90 // Get Gauss offset and face's owner patch's ID
91 gaussOffset_.setSize(nFaces_);
92 ownerPatchIDs_.setSize(nFaces_);
93 for (label f = 0; f < nFaces_; ++f)
94 {
95 gaussOffset_[f] = f * nGaussPerFace_;
96
97 if (faces_[f]->isBoundary())
98 {
99 ownerPatchIDs_[f] = faces_[f]->patchID();
100 }
101 else
102 {
103 ownerPatchIDs_[f] = -1;
104 }
105 }
106
107
108 plusValues_.setSize(nGauss_);
109 minusValues_.setSize(nGauss_);
110
111 interpolateFromDof();
112}

328template<class Type>
329void Foam::faceGaussField<Type>::interpolateFromDof()
330{
331 const cellDof<Type>& cellPDof = *cellsDof_[0];
332
333 for (label localFaceI = 0; localFaceI < nFaces_; ++localFaceI)
334 {
335 const label faceID = (*facesID_)[localFaceI];
336 const label owner = mesh_->faceOwner()[faceID];
337 const bool isOwner = (owner == cellID_);
338 bool isBoundaryFace = faces_[localFaceI]->isBoundary();
339
340 const dgGeomFace& face = *faces_[localFaceI];
341
342 // Get Gauss points for owner sides
343 const List<vector>& gpOwner = face.gaussPointsOwner();
344
345 // Get basis functions for owner sides
346 const List<List<scalar>>& ownerBasis = face.ownerBasis();
347
348 const label offset = gaussOffset_[localFaceI];
349
350 if (isOwner)
351 {
352 forAll(gpOwner, i)
353 {
354 minusValues_[offset + i] = pTraits<Type>::zero;
355 for (label k = 0; k < cellPDof.nDof(); ++k)
356 {
357 minusValues_[offset + i] += cellPDof[k] * ownerBasis[i][k];
358 }
359 }
360
361 if (isBoundaryFace)
362 {
363 forAll(gpOwner, i)
364 {
365 plusValues_[offset + i] = minusValues_[offset + i]; //pTraits<Type>::max; //pTraits<Type>::zero;
366 }
367 }
368 else
369 {
370 // Get Gauss points for neighbor sides
371 const List<vector>& gpNeigh = face.gaussPointsNeighbor();
372
373 // Get basis functions for neighbor sides
374 const List<List<scalar>>& neighborBasis = face.neighborBasis();
375
376 const cellDof<Type>& cellNDof = *cellsDof_[localFaceI + 1];
377 forAll(gpNeigh, i)
378 {
379 plusValues_[offset + i] = pTraits<Type>::zero;
380 for (label k = 0; k < cellNDof.nDof(); ++k)
381 {
382 plusValues_[offset + i] += cellNDof[k] * neighborBasis[i][k];
383 }
384 }
385 }
386 }
387 else
388 {
389 // This face has both owner and neighbor sides, so it is internal
390
391 // Get Gauss points for neighbor sides
392 const List<vector>& gpNeigh = face.gaussPointsNeighbor();
393
394 // Get basis functions for neighbor sides
395 const List<List<scalar>>& neighborBasis = face.neighborBasis();
396
397 const cellDof<Type>& cellNDof = *cellsDof_[localFaceI + 1];
398
399 forAll(gpNeigh, i)
400 {
401 minusValues_[offset + i] = pTraits<Type>::zero;
402 for (label k = 0; k < cellPDof.nDof(); ++k)
403 {
404 minusValues_[offset + i] += cellPDof[k] * neighborBasis[i][k];
405 }
406 }
407
408 forAll(gpOwner, i)
409 {
410 plusValues_[offset + i] = pTraits<Type>::zero;
411 for (label k = 0; k < cellNDof.nDof(); ++k)
412 {
413 plusValues_[offset + i] += cellNDof[k] * ownerBasis[i][k];
414 }
415 }
416 }
417 }
418}
