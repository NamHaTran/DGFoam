45// * * * * * * * * Compute scalar basis * * * * * * * //
46
47inline void dgBasisField::computeBasisField()
48{
49 const dgGeomCell& cell = *mesh_.cells()[cellID_];
50 const List<List<scalar>>& basisCell = cell.basis();
51
52 const label nFaces = faceLabels_.size();
53 basis_.setSize(nDof_);
54
55 for (label dof = 0; dof < nDof_; ++dof)
56 {
57 // Create GaussField for basis[dof]
58 GaussField<scalar> phi(cellID_, &mesh_);
59
60 // ---- CELL GAUSS POINTS ---- //
61 for (label g = 0; g < cell.nGauss(); ++g)
62 {
63 phi.cellValueAt(g) = basisCell[g][dof];
64 }
65
66 // ---- FACE GAUSS POINTS ---- //
67 for (label lf = 0; lf < nFaces; ++lf)
68 {
69 const label faceID = faceLabels_[lf];
70 const dgGeomFace& face = *mesh_.faces()[faceID];
71 const bool isBoundary = (faceID >= mesh_.nInternalFaces());
72 const bool ownerSide = (mesh_.faceOwner()[faceID] == cellID_);
73
74 // OWNER SIDE CASES
75 if (ownerSide)
76 {
77 const auto& minusB = face.ownerBasis();
78 const auto& plusB =
79 isBoundary ? minusB : face.neighborBasis();
80
81 const label nG = face.gaussPointsOwner().size();
82
83 for (label g = 0; g < nG; ++g)
84 {
85 phi.faceMinusValueAt(lf, g) = minusB[g][dof];
86 phi.facePlusValueAt(lf, g) = plusB[g][dof];
87 }
88 }
89 else // NEIGHBOR SIDE CASES
90 {
91 const auto& minusB = face.neighborBasis();
92 const auto& plusB = face.ownerBasis();
93 const label nG = face.gaussPointsNeighbor().size();
94
95 for (label g = 0; g < nG; ++g)
96 {
97 phi.faceMinusValueAt(lf, g) = minusB[g][dof];
98 phi.facePlusValueAt(lf, g) = plusB[g][dof];
99 }
100 }
101 }
102
103 basis_[dof] = phi;
104 }
105}

108// * * * * * * * Compute derivative basis * * * * * * * //
109
110inline void dgBasisField::computeDBasisField()
111{
112 const dgGeomCell& cell = *mesh_.cells()[cellID_];
113
114 const auto& dB1 = cell.dBasis_dEta1();
115 const auto& dB2 = cell.dBasis_dEta2();
116 const auto& dB3 = cell.dBasis_dEta3();
117
118 const label nFaces = faceLabels_.size();
119 dBasis_.setSize(nDof_);
120
121 for (label dof = 0; dof < nDof_; ++dof)
122 {
123 GaussField<vector> dPhi(cellID_, &mesh_);
124
125 // ---- CELL GAUSS POINTS ---- //
126 for (label g = 0; g < cell.nGauss(); ++g)
127 {
128 dPhi.cellValueAt(g) = vector(dB1[g][dof], dB2[g][dof], dB3[g][dof]);
129 }
130
131 // ---- FACE GAUSS POINTS ---- //
132 for (label lf = 0; lf < nFaces; ++lf)
133 {
134 const label faceID = faceLabels_[lf];
135 const dgGeomFace& face = *mesh_.faces()[faceID];
136 const bool isBoundary = (faceID >= mesh_.nInternalFaces());
137 const bool ownerSide = (mesh_.faceOwner()[faceID] == cellID_);
138
139 auto loadVec = [&](const List<List<scalar>>& B1,
140 const List<List<scalar>>& B2,
141 const List<List<scalar>>& B3,
142 label g)
143 {
144 return vector(B1[g][dof], B2[g][dof], B3[g][dof]);
145 };
146
147 if (ownerSide)
148 {
149 const auto& m1 = face.owner_dBasis_dEta1();
150 const auto& m2 = face.owner_dBasis_dEta2();
151 const auto& m3 = face.owner_dBasis_dEta3();
152
153 const auto& p1 =
154 isBoundary ? m1 : face.neighbor_dBasis_dEta1();
155 const auto& p2 =
156 isBoundary ? m2 : face.neighbor_dBasis_dEta2();
157 const auto& p3 =
158 isBoundary ? m3 : face.neighbor_dBasis_dEta3();
159
160 const label nG = face.gaussPointsOwner().size();
161
162 for (label g = 0; g < nG; ++g)
163 {
164 dPhi.faceMinusValueAt(lf, g) = loadVec(m1, m2, m3, g);
165 dPhi.facePlusValueAt(lf, g) = loadVec(p1, p2, p3, g);
166 }
167 }
168 else
169 {
170 const auto& m1 = face.neighbor_dBasis_dEta1();
171 const auto& m2 = face.neighbor_dBasis_dEta2();
172 const auto& m3 = face.neighbor_dBasis_dEta3();
173
174 const auto& p1 = face.owner_dBasis_dEta1();
175 const auto& p2 = face.owner_dBasis_dEta2();
176 const auto& p3 = face.owner_dBasis_dEta3();
177
178 const label nG = face.gaussPointsNeighbor().size();
179
180 for (label g = 0; g < nG; ++g)
181 {
182 dPhi.faceMinusValueAt(lf, g) = loadVec(m1, m2, m3, g);
183 dPhi.facePlusValueAt(lf, g) = loadVec(p1, p2, p3, g);
184 }
185 }
186 }
187
188 dBasis_[dof] = dPhi;
189 }
190}
