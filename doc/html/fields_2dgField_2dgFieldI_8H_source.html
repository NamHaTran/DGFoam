49template<class Type>
50dgField<Type>::dgField
51(
52 const IOobject& io,
53 const dgGeomMesh& mesh,
54 bool hasDof
55)
56:
57 regIOobject(io),
58 mesh_(mesh),
59 hasDof_(hasDof),
60 dofPtr_(nullptr)
61{
62 const label nCells = mesh_.nCells();
63 gaussFields_.setSize(nCells);
64
65 if (hasDof)
66 {
67 // Create DOF container
68 dofPtr_.reset
69 (
70 new dofField<Type>
71 (
72 io.name(),
73 io.instance(),
74 io.readOpt(),
75 io.writeOpt(),
76 mesh
77 )
78 );
79
80 // Interpolate DOF â†’ Gauss values
81 for (label cellI = 0; cellI < nCells; ++cellI)
82 {
83 gaussFields_[cellI] = GaussField<Type>(dofPtr_, cellI, &mesh_);
84 gaussFields_[cellI].interpolateFromDof();
85 }
86
87 Info<< "dgField<" << pTraits<Type>::typeName
88 << ">: constructed conservative field " << io.name()
89 << " with " << nCells << " cells." << nl;
90 }
91 else
92 {
93 // Primary field: allocate Gauss structure only
94 for (label cellI = 0; cellI < nCells; ++cellI)
95 {
96 gaussFields_[cellI] = GaussField<Type>(cellI, &mesh_);
97 }
98
99 Info<< "dgField<" << pTraits<Type>::typeName
100 << ">: constructed primary field " << io.name()
101 << " (no DOF)." << nl;
102 }
103}
