51template<class Type>
52dgField<Type>::dgField
53(
54 const IOobject& io,
55 const dgGeomMesh& mesh,
56 bool hasDof
57)
58:
59 regIOobject(io),
60 mesh_(mesh),
61 hasDof_(hasDof),
62 dofPtr_(nullptr),
63 procSyncPtr_(nullptr)
64{
65 const label nCells = mesh_.nCells();
66 gaussFields_.setSize(nCells);
67
68 if (hasDof)
69 {
70 // Create DOF container
71 dofPtr_.reset
72 (
73 new dofField<Type>
74 (
75 io.name(),
76 io.instance(),
77 io.readOpt(),
78 io.writeOpt(),
79 mesh
80 )
81 );
82
83 // Interpolate DOF → Gauss values
84 for (label cellI = 0; cellI < nCells; ++cellI)
85 {
86 gaussFields_[cellI] = GaussField<Type>(dofPtr_, cellI, &mesh_);
87 gaussFields_[cellI].interpolateFromDof();
88 }
89
90 //Info<< "dgField<" << pTraits<Type>::typeName
91 // << ">: constructed conservative field " << io.name()
92 // << " with " << nCells << " cells." << nl;
93 }
94 else
95 {
96 // Primary field: allocate Gauss structure only
97 for (label cellI = 0; cellI < nCells; ++cellI)
98 {
99 gaussFields_[cellI] = GaussField<Type>(cellI, &mesh_);
100 }
101
102 //Info<< "dgField<" << pTraits<Type>::typeName
103 // << ">: constructed primary field " << io.name()
104 // << " (no DOF)." << nl;
105 }
106}

121template<class Type>
122tmp<GeometricField<Type, fvPatchField, volMesh>>
123dgField<Type>::cellAverageVolField() const
124{
125 tmp<GeometricField<Type, fvPatchField, volMesh>> tAvgField
126 (
127 new GeometricField<Type, fvPatchField, volMesh>
128 (
129 IOobject
130 (
131 this->name(),
132 this->instance(),
133 mesh_.getFvMesh(),
134 IOobject::NO_READ,
135 IOobject::NO_WRITE,
136 false
137 ),
138 mesh_.getFvMesh(),
139 dimensioned<Type>("zero", dimless, pTraits<Type>::zero)
140 )
141 );
142 auto& avgField = tAvgField.ref();
143
144 forAll(gaussFields_, cellI)
145 {
146 const dgGeomCell& cell = *mesh_.cells()[cellI];
147 const cellGaussField<Type>& cellField = gaussFields_[cellI].cellField();
148 const List<scalar>& weights = cell.weights();
149 const List<scalar>& J3D = cell.J3D();
150
151 Type integral = pTraits<Type>::zero;
152 scalar totalWeight = 0.0;
153
154 forAll(weights, gpI)
155 {
156 scalar refJacobian = 1.0;
157
158 switch (cell.type())
159 {
160 case dgCellType::HEX:
161 break;
162
163 case dgCellType::PRISM:
164 refJacobian =
165 referenceJacobian::prismRefToHexRef(cell.gaussPoints()[gpI]);
166 break;
167
168 case dgCellType::PYRAMID:
169 refJacobian =
170 referenceJacobian::pyramidRefToHexRef(cell.gaussPoints()[gpI]);
171 break;
172
173 case dgCellType::TET:
174 refJacobian =
175 referenceJacobian::tetRefToHexRef(cell.gaussPoints()[gpI]);
176 break;
177
178 default:
179 FatalErrorInFunction
180 << "Unsupported cell type for cell-average write in cell "
181 << cellI << nl
182 << abort(FatalError);
183 }
184
185 const scalar quadWeight = weights[gpI]*J3D[gpI]*refJacobian;
186 integral += cellField[gpI]*quadWeight;
187 totalWeight += quadWeight;
188 }
189
190 if (mag(totalWeight) <= SMALL)
191 {
192 FatalErrorInFunction
193 << "Non-positive quadrature weight while writing averaged field "
194 << this->name() << " in cell " << cellI << nl
195 << abort(FatalError);
196 }
197
198 avgField[cellI] = integral/totalWeight;
199 }
200
201 return tAvgField;
202}
