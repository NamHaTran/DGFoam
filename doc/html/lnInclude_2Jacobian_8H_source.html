69inline Foam::tensor HexJacobian
70(
71 const vector& gaussPt, // Gauss point in reference space (eta)
72 const List<vector>& cellVertices // Physical coordinates of hex vertices
73)
74{
75 // ---------------------------------------------------------------------
76 // Extract reference coordinates (eta-space)
77 // ---------------------------------------------------------------------
78 const scalar eta1 = gaussPt.x();
79 const scalar eta2 = gaussPt.y();
80 const scalar eta3 = gaussPt.z();
81
82 // ---------------------------------------------------------------------
83 // Gradients of shape functions w.r.t eta (dPhi/dEta)
84 // Ordering: A B C D E F G H
85 // ---------------------------------------------------------------------
86 List<vector> gradPhi_Eta(cellVertices.size(), Zero);
87
88 // Vertex A
89 gradPhi_Eta[0].x() = -(1 - eta2)*(1 - eta3)/8;
90 gradPhi_Eta[0].y() = -(1 - eta1)*(1 - eta3)/8;
91 gradPhi_Eta[0].z() = -(1 - eta1)*(1 - eta2)/8;
92
93 // Vertex B
94 gradPhi_Eta[1].x() = (1 - eta2)*(1 - eta3)/8;
95 gradPhi_Eta[1].y() = -(1 + eta1)*(1 - eta3)/8;
96 gradPhi_Eta[1].z() = -(1 + eta1)*(1 - eta2)/8;
97
98 // Vertex C
99 gradPhi_Eta[2].x() = -(1 + eta2)*(1 - eta3)/8;
100 gradPhi_Eta[2].y() = (1 - eta1)*(1 - eta3)/8;
101 gradPhi_Eta[2].z() = -(1 - eta1)*(1 + eta2)/8;
102
103 // Vertex D
104 gradPhi_Eta[3].x() = (1 + eta2)*(1 - eta3)/8;
105 gradPhi_Eta[3].y() = (1 + eta1)*(1 - eta3)/8;
106 gradPhi_Eta[3].z() = -(1 + eta1)*(1 + eta2)/8;
107
108 // Vertex E
109 gradPhi_Eta[4].x() = -(1 - eta2)*(1 + eta3)/8;
110 gradPhi_Eta[4].y() = -(1 - eta1)*(1 + eta3)/8;
111 gradPhi_Eta[4].z() = (1 - eta1)*(1 - eta2)/8;
112
113 // Vertex F
114 gradPhi_Eta[5].x() = (1 - eta2)*(1 + eta3)/8;
115 gradPhi_Eta[5].y() = -(1 + eta1)*(1 + eta3)/8;
116 gradPhi_Eta[5].z() = (1 + eta1)*(1 - eta2)/8;
117
118 // Vertex G
119 gradPhi_Eta[6].x() = -(1 + eta2)*(1 + eta3)/8;
120 gradPhi_Eta[6].y() = (1 - eta1)*(1 + eta3)/8;
121 gradPhi_Eta[6].z() = (1 - eta1)*(1 + eta2)/8;
122
123 // Vertex H
124 gradPhi_Eta[7].x() = (1 + eta2)*(1 + eta3)/8;
125 gradPhi_Eta[7].y() = (1 + eta1)*(1 + eta3)/8;
126 gradPhi_Eta[7].z() = (1 + eta1)*(1 + eta2)/8;
127
128 // ---------------------------------------------------------------------
129 // Compute physical gradients: d(x,y,z)/d(eta1,eta2,eta3)
130 // Each vector stores partial derivatives w.r.t eta
131 // ---------------------------------------------------------------------
132 vector gradX1_Eta(Zero); // d(x)/d(eta)
133 vector gradX2_Eta(Zero); // d(y)/d(eta)
134 vector gradX3_Eta(Zero); // d(z)/d(eta)
135
136 for (label i = 0; i < cellVertices.size(); ++i)
137 {
138 gradX1_Eta.x() += gradPhi_Eta[i].x()*cellVertices[i].x();
139 gradX1_Eta.y() += gradPhi_Eta[i].y()*cellVertices[i].x();
140 gradX1_Eta.z() += gradPhi_Eta[i].z()*cellVertices[i].x();
141
142 gradX2_Eta.x() += gradPhi_Eta[i].x()*cellVertices[i].y();
143 gradX2_Eta.y() += gradPhi_Eta[i].y()*cellVertices[i].y();
144 gradX2_Eta.z() += gradPhi_Eta[i].z()*cellVertices[i].y();
145
146 gradX3_Eta.x() += gradPhi_Eta[i].x()*cellVertices[i].z();
147 gradX3_Eta.y() += gradPhi_Eta[i].y()*cellVertices[i].z();
148 gradX3_Eta.z() += gradPhi_Eta[i].z()*cellVertices[i].z();
149 }
150
151 // ---------------------------------------------------------------------
152 // Jacobian tensor (columns = gradients of x, y, z w.r.t eta)
153 // ---------------------------------------------------------------------
154 const Foam::tensor Jacobian
155 (
156 gradX1_Eta,
157 gradX2_Eta,
158 gradX3_Eta
159 );
160
161 return Jacobian;
162}

170inline Foam::tensor PrismJacobian
171(
172 const vector& gaussPt, // Gauss point in reference space (eta)
173 const List<vector>& cellVertices // Physical coordinates of prism vertices
174)
175{
176 // ---------------------------------------------------------------------
177 // Extract reference coordinates (eta-space)
178 // ---------------------------------------------------------------------
179 const scalar eta1 = gaussPt.x();
180 const scalar eta2 = gaussPt.y();
181 const scalar eta3 = gaussPt.z();
182
183 // ---------------------------------------------------------------------
184 // Gradients of shape functions w.r.t eta (dPhi/dEta)
185 //
186 // Vertex correspondence:
187 //
188 // cellVertices id -> prism vertex -> gradPhi_Eta id
189 // -------------------------------------------------
190 // 0 -> E -> 0
191 // 1 -> A -> 1
192 // 2 -> B -> 2
193 // 3 -> F -> 3
194 // 4 -> D -> 4
195 // 5 -> C -> 5
196 //
197 // ---------------------------------------------------------------------
198 List<vector> gradPhi_Eta(cellVertices.size(), Zero);
199
200 // ---------------------------------------------------------------------
201 // Vertex A (cellVertices[1])
202 // ---------------------------------------------------------------------
203 gradPhi_Eta[1].x() = -(1 - eta2)*(1 - eta3)/8;
204 gradPhi_Eta[1].y() = -(1 - eta1)*(1 - eta3)/8;
205 gradPhi_Eta[1].z() = -(1 - eta1)*(1 - eta2)/8;
206
207 // ---------------------------------------------------------------------
208 // Vertex B (cellVertices[2])
209 // ---------------------------------------------------------------------
210 gradPhi_Eta[2].x() = (1 - eta2)*(1 - eta3)/8;
211 gradPhi_Eta[2].y() = -(1 + eta1)*(1 - eta3)/8;
212 gradPhi_Eta[2].z() = -(1 + eta1)*(1 - eta2)/8;
213
214 // ---------------------------------------------------------------------
215 // Vertex C (cellVertices[5])
216 // ---------------------------------------------------------------------
217 gradPhi_Eta[5].x() = -(1 + eta2)*(1 - eta3)/8;
218 gradPhi_Eta[5].y() = (1 - eta1)*(1 - eta3)/8;
219 gradPhi_Eta[5].z() = -(1 - eta1)*(1 + eta2)/8;
220
221 // ---------------------------------------------------------------------
222 // Vertex D (cellVertices[4])
223 // ---------------------------------------------------------------------
224 gradPhi_Eta[4].x() = (1 + eta2)*(1 - eta3)/8;
225 gradPhi_Eta[4].y() = (1 + eta1)*(1 - eta3)/8;
226 gradPhi_Eta[4].z() = -(1 + eta1)*(1 + eta2)/8;
227
228 // ---------------------------------------------------------------------
229 // Vertex E (cellVertices[0])
230 // ---------------------------------------------------------------------
231 gradPhi_Eta[0].x() = 0;
232 gradPhi_Eta[0].y() = -(1 + eta3)/4;
233 gradPhi_Eta[0].z() = (1 - eta2)/4;
234
235 // ---------------------------------------------------------------------
236 // Vertex F (cellVertices[3])
237 // ---------------------------------------------------------------------
238 gradPhi_Eta[3].x() = 0;
239 gradPhi_Eta[3].y() = (1 + eta3)/4;
240 gradPhi_Eta[3].z() = (1 + eta2)/4;
241
242 // ---------------------------------------------------------------------
243 // Compute physical gradients: d(x,y,z)/d(eta1,eta2,eta3)
244 // ---------------------------------------------------------------------
245 vector gradX1_Eta(Zero); // d(x)/d(eta)
246 vector gradX2_Eta(Zero); // d(y)/d(eta)
247 vector gradX3_Eta(Zero); // d(z)/d(eta)
248
249 for (label i = 0; i < cellVertices.size(); ++i)
250 {
251 gradX1_Eta.x() += gradPhi_Eta[i].x()*cellVertices[i].x();
252 gradX1_Eta.y() += gradPhi_Eta[i].y()*cellVertices[i].x();
253 gradX1_Eta.z() += gradPhi_Eta[i].z()*cellVertices[i].x();
254
255 gradX2_Eta.x() += gradPhi_Eta[i].x()*cellVertices[i].y();
256 gradX2_Eta.y() += gradPhi_Eta[i].y()*cellVertices[i].y();
257 gradX2_Eta.z() += gradPhi_Eta[i].z()*cellVertices[i].y();
258
259 gradX3_Eta.x() += gradPhi_Eta[i].x()*cellVertices[i].z();
260 gradX3_Eta.y() += gradPhi_Eta[i].y()*cellVertices[i].z();
261 gradX3_Eta.z() += gradPhi_Eta[i].z()*cellVertices[i].z();
262 }
263
264 // ---------------------------------------------------------------------
265 // Jacobian tensor (columns = gradients of x, y, z w.r.t eta)
266 // ---------------------------------------------------------------------
267 const Foam::tensor Jacobian
268 (
269 gradX1_Eta,
270 gradX2_Eta,
271 gradX3_Eta
272 );
273
274 return Jacobian;

486inline Foam::scalar calcJacobianDetAtInteriorGaussPt
487(
488 dgCellType cellType,
489 const vector& gaussPt, // Gauss point in reference space (eta)
490 const List<vector>& cellVertices // Physical coordinates of tetrahedron vertices
491)
492{
493 Foam::tensor J;
494
495 switch (cellType)
496 {
497 case dgCellType::HEX:
498 J = HexJacobian(gaussPt, cellVertices);
499 break;
500
501 case dgCellType::PRISM:
502 J = PrismJacobian(gaussPt, cellVertices);
503 break;
504
505 case dgCellType::PYRAMID:
506 J = PyramidJacobian(gaussPt, cellVertices);
507 break;
508
509 case dgCellType::TET:
510 J = TetraJacobian(gaussPt, cellVertices);
511 break;
512
513 default:
514 FatalErrorInFunction
515 << "Unsupported cell type for 3D Jacobian calculation: "
516 << static_cast<int>(cellType) << abort(FatalError);
517 }
518
519 return det(J);
520}

530inline Foam::scalar calcLameParam
531(
532 dgCellType cellType,
533 dgFacePosition facePos,
534 const vector& gaussPt, // Gauss point in reference space (eta)
535 const List<vector>& cellVertices // Physical coordinates of vertices
536)
537{
538 // Caclulate Jacobian tensor
539 Foam::tensor J;
540
541 switch (cellType)
542 {
543 case dgCellType::HEX:
544 J = HexJacobian(gaussPt, cellVertices);
545 break;
546
547 case dgCellType::PRISM:
548 J = PrismJacobian(gaussPt, cellVertices);
549 break;
550
551 case dgCellType::PYRAMID:
552 J = PyramidJacobian(gaussPt, cellVertices);
553 break;
554
555 case dgCellType::TET:
556 J = TetraJacobian(gaussPt, cellVertices);
557 break;
558
559 default:
560 FatalErrorInFunction
561 << "Unsupported cell type for 3D Jacobian calculation: "
562 << static_cast<int>(cellType) << abort(FatalError);
563 }
564
565 // Compute Lame parameter based on face position
566 scalar lameParam = 0.0;
567 switch (facePos)
568 {
569 case dgFacePosition::ABCD:
570 case dgFacePosition::EFGH:
571 {
572 // eta3 direction, eta3 = +/-1
573 vector gradX_eta1 (J.xx(), J.yx(), J.zx());
574 vector gradX_eta2 (J.xy(), J.yy(), J.zy());
575 lameParam = mag(gradX_eta1 ^ gradX_eta2);
576 break;
577 }
578 case dgFacePosition::ABEF:
579 case dgFacePosition::CDGH:
580 {
581 // eta2 direction, eta2 = +/-1
582 vector gradX_eta1 (J.xx(), J.yx(), J.zx());
583 vector gradX_eta3 (J.xz(), J.yz(), J.zz());
584 lameParam = mag(gradX_eta3 ^ gradX_eta1);
585 break;
586 }
587 case dgFacePosition::ACEG:
588 case dgFacePosition::BDFH:
589 {
590 // eta1 direction, eta1 = +/-1
591 vector gradX_eta2 (J.xy(), J.yy(), J.zy());
592 vector gradX_eta3 (J.xz(), J.yz(), J.zz());
593 lameParam = mag(gradX_eta2 ^ gradX_eta3);
594 break;
595 }
596 case dgFacePosition::NONE:
597 {
598 FatalErrorInFunction
599 << "Cannot compute Lame parameter for face with NONE position."
600 << abort(FatalError);
601 break;
602 }
603 }
604
605 return lameParam;
606}

634 inline Foam::tensor prismRefJMontjoieToDG(const Foam::vector& etaPt)
635 {
636 Foam::vector xiPt = etaToXiPrism(etaPt);
637
638 Foam::scalar xi1 = xiPt.x();
639 Foam::scalar xi3 = xiPt.z();
640
641 Foam::scalar eta1 = etaPt.x();
642 Foam::scalar eta3 = etaPt.z();
643
644 Foam::tensor JxXi(Zero);
645 JxXi.xx() = (1 - xi3)/4;
646 JxXi.xz() = - (xi1 + 1)/4;
647 JxXi.yz() = 0.5;
648 JxXi.zy() = 0.5;
649
650 Foam::tensor JxiEta(Zero);
651 JxiEta.xx() = (1 - eta3)/2;
652 JxiEta.xz() = -0.5*(1 + eta1);
653 JxiEta.yy() = 1.0;
654 JxiEta.zz() = 1.0;
655
656 return (JxiEta.T() & JxXi.T());
657 };

666 inline Foam::tensor pyramidRefJMontjoieToDG(const Foam::vector& etaPt)
667 {
668 Foam::vector xiPt = etaToXiPyramid(etaPt);
669
670 Foam::scalar xi1 = xiPt.x();
671 Foam::scalar xi2 = xiPt.y();
672 Foam::scalar xi3 = xiPt.z();
673
674 Foam::scalar eta1 = etaPt.x();
675 Foam::scalar eta2 = etaPt.y();
676 Foam::scalar eta3 = etaPt.z();
677
678 Foam::tensor JxXi(Zero);
679 JxXi.xx() = (1 - xi3)/2;
680 JxXi.xy() = 0;
681 JxXi.xz() = - xi1/2;
682 JxXi.yx() = 0;
683 JxXi.yy() = (1 - xi3)/2;
684 JxXi.yz() = - xi2/2;
685 JxXi.zz() = 0.5;
686
687 Foam::tensor JxiEta(Zero);
688 JxiEta.xx() = (1 - eta3)/2;
