125 virtual void updateGhostStateValue
126 (
127 const label cellID,
128 const label faceLocalID,
129 const label localGauss,
130 const vector& n,
131 const Type& minusValue,
132 Type& plusValue
133 ) const override
134 {
135 const vector& Un =
136 U_.gaussFields()[cellID].facePlusValueAt(faceLocalID, localGauss);
137
138 const scalar dotProd = Un & n;
139
140 if (dotProd < 0) // Inlet: flow into domain
141 {
142 plusValue = 2 * inletValue_ - minusValue;
143 }
144 else // Outlet: flow out of domain
145 {
146 plusValue = minusValue;
147 }
148 }
