156 int genSendTag() const
157 {
158 const label myProc = Pstream::myProcNo();
159
160 if (myProc >= 200000 || neighbProc_ >= 200000)
161 {
162 FatalErrorInFunction
163 << "Processor rank too large for tag scheme: "
164 << myProc << " " << neighbProc_
165 << exit(FatalError);
166 }
167
168 return myProc*1e6 + neighbProc_;
169 }
