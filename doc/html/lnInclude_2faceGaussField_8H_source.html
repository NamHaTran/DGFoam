27Class
28 Foam::faceGaussField

30Description
31 Gauss-point field values on faces of a DG cell.
32
33 Stores values at Gauss integration points on all faces of a cell,
34 including both minus (cell-local interior) and plus (exterior) sides.

473 inline const cellDof<Type>& neighbourCellDof
474 (
475 const label localFaceI
476 ) const
477 {
478 if (localFaceI < 0 || localFaceI >= nFaces_)
479 {
480 FatalErrorInFunction
481 << "Face index " << localFaceI << " is out of range [0, "
482 << nFaces_ - 1 << "] for cell " << cellID_
483 << abort(FatalError);
484 }
485
486 const dgGeomFace& face = *faces_[localFaceI];
487
488 if (!face.isBoundary())
489 {
490 const cellDof<Type>* nbrDofPtr = cellsDof_[localFaceI + 1];
491
492 if (!nbrDofPtr)
493 {
494 FatalErrorInFunction
495 << "Missing neighbour DoF pointer on internal face "
496 << localFaceI << " of cell " << cellID_
497 << abort(FatalError);
498 }
499
500 return *nbrDofPtr;
501 }
502
503 if (!face.isProcessorPatch())
504 {
505 FatalErrorInFunction
506 << "Face " << localFaceI << " of cell " << cellID_
507 << " is a physical boundary face and has no neighbour DoF."
508 << abort(FatalError);
509 }
510
511 if (!bufferedNbrDofValid_[localFaceI])
512 {
513 FatalErrorInFunction
514 << "Processor neighbour DoF buffer is not initialized for "
515 << "face " << localFaceI << " of cell " << cellID_
516 << ". Synchronize processor boundaries before accessing "
517 << "neighbourCellDof()."
518 << abort(FatalError);
519 }
520
521 return bufferedNbrDof_[localFaceI];
522 }
