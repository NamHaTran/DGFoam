27Description
28 A collection of utilities for transforming between different reference
29 coordinate systems.
30
31 DGFoam uses three reference coordinate systems:
32 - DG hexahedral reference coordinate system (eta-space):
33 used for numerical integration.
34 - DG element reference coordinate system (xi-space):
35 serves as an intermediate coordinate system between eta-space
36 and the Montjoie reference space.
37 - Montjoie reference coordinate system (x-space):
38 used for the evaluation of basis functions.

173 inline Foam::vector xToXiPrism(const Foam::vector& xPt)
174 {
175 const Foam::scalar x = xPt.x();
176 const Foam::scalar y = xPt.y();
177 const Foam::scalar z = xPt.z();
178
179 Foam::vector xiPt = xPt;
180
181 // xi3 from y
182 xiPt.z() = 2.0 * y - 1.0;
183
184 // xi2 from z
185 xiPt.y() = 2.0 * z - 1.0;
186
187 // xi1 from x and y (using 1 - xi3 = 2(1 - y))
188 xiPt.x() = 2.0 * x / (1.0 - y) - 1.0;
189
190 return xiPt;
191 };

262 inline Foam::vector xToXiPyramid(const Foam::vector& xPt)
263 {
264 const Foam::scalar x = xPt.x();

266 const Foam::scalar z = xPt.z();

286 inline Foam::vector etaToXiTet(const Foam::vector& etaPt)
287 {
288 const Foam::scalar eta1 = etaPt.x();
289 const Foam::scalar eta2 = etaPt.y();
290 const Foam::scalar eta3 = etaPt.z();
291
292 Foam::vector xiPt = etaPt;
293
294 // xi3 = eta3
295 xiPt.z() = eta3;
296
297 // xi2 from eta2 and eta3
298 // eta2 + 1 = 2*(xi2 + 1)/(1 - xi3)
299 xiPt.y() = 0.5 * (eta2 + 1.0) * (1.0 - eta3) - 1.0;
300
301 // xi1 from eta1, eta2, eta3
302 // eta1 + 1 = 2*(xi1 + 1)/(-xi2 - xi3)
303 // with: -xi2 - xi3 = (1 - eta3)(1 - eta2)/2
304 xiPt.x() =
305 0.25 * (eta1 + 1.0) * (1.0 - eta3) * (1.0 - eta2) - 1.0;
306
307 return xiPt;
308 };

317 inline Foam::vector xiToEtaTet(const Foam::vector& xiPt)
318 {
319 Foam::scalar xi1 = xiPt.x();
320 Foam::scalar xi2 = xiPt.y();
321 Foam::scalar xi3 = xiPt.z();
322
323 Foam::vector etaPt = xiPt;
324 etaPt.x() = 2.0 * (xi1 + 1.0)/(-xi2 - xi3) - 1.0;
325 etaPt.y() = 2.0 * (xi2 + 1.0)/(1 - xi3) - 1.0;
326
327 return etaPt;
328 };
