29 coordinate systems.
30
31 DGFoam uses three reference coordinate systems:
32 - DG hexahedral reference coordinate system (eta-space):
33 used for numerical integration.
34 - DG element reference coordinate system (xi-space):

154 inline Foam::vector etaToXiTet(const Foam::vector& etaPt)
155 {
156 const Foam::scalar eta1 = etaPt.x();
157 const Foam::scalar eta2 = etaPt.y();
158 const Foam::scalar eta3 = etaPt.z();
159
160 Foam::vector xiPt = etaPt;
161
162 // xi3 = eta3
163 xiPt.z() = eta3;
164
165 // xi2 from eta2 and eta3
166 // eta2 + 1 = 2*(xi2 + 1)/(1 - xi3)
167 xiPt.y() = 0.5 * (eta2 + 1.0) * (1.0 - eta3) - 1.0;
168
169 // xi1 from eta1, eta2, eta3
170 // eta1 + 1 = 2*(xi1 + 1)/(-xi2 - xi3)
171 // with: -xi2 - xi3 = (1 - eta3)(1 - eta2)/2
172 xiPt.x() =
173 0.25 * (eta1 + 1.0) * (1.0 - eta3) * (1.0 - eta2) - 1.0;
174
175 return xiPt;
176 };

185 inline Foam::vector xiToEtaTet(const Foam::vector& xiPt)
186 {
187 Foam::scalar xi1 = xiPt.x();
188 Foam::scalar xi2 = xiPt.y();
189 Foam::scalar xi3 = xiPt.z();
190
191 Foam::vector etaPt = xiPt;
192 etaPt.x() = 2.0 * (xi1 + 1.0)/(-xi2 - xi3) - 1.0;
193 etaPt.y() = 2.0 * (xi2 + 1.0)/(1 - xi3) - 1.0;
194
195 return etaPt;
196 };

214 inline vector2D squareToTriangle
215 (
216 const vector2D& etaPt
217 )
218 {
219 const scalar eta1 = etaPt.x();
220 const scalar eta2 = etaPt.y();
221
222 // ---------------------------------------------------------------------
223 // Map square [-1,1] -> [0,1]
224 // ---------------------------------------------------------------------
225
226 const scalar a = 0.5*(1.0 + eta1);
227 const scalar b = 0.5*(1.0 + eta2);
228
229 // ---------------------------------------------------------------------
230 // Duffy transform
231 // ---------------------------------------------------------------------
232
233 const scalar r = a;
234 const scalar s = b*(1.0 - a);
235
236 return vector2D(r, s);
237 }
