61inline label getNumBasis(const label pOrder, const dgCellType type)
62{
63 label count = 0;
64
65 switch (type)
66 {
67 // --------------------------------------------------
68 // Hexahedron
69 // Constraint: i + j + k <= r
70 // --------------------------------------------------
71 case dgCellType::HEX:
72 {
73 for (label i = 0; i <= pOrder; ++i)
74 for (label j = 0; j <= pOrder - i; ++j)
75 for (label k = 0; k <= pOrder - i - j; ++k)
76 ++count;
77
78 return count;
79 }
80
81 // --------------------------------------------------
82 // Prism (wedge)
83 // Constraint: i + j <= r, k <= r
84 // --------------------------------------------------
85 case dgCellType::PRISM:
86 {
87 for (label i = 0; i <= pOrder; ++i)
88 for (label j = 0; j <= pOrder - i; ++j)
89 for (label k = 0; k <= pOrder; ++k)
90 ++count;
91
92 return count;
93 }
94
95 // --------------------------------------------------
96 // Pyramid
97 // Constraint: 0 <= i,j <= r - k, k <= r
98 // --------------------------------------------------
99 case dgCellType::PYRAMID:
100 {
101 for (label k = 0; k <= pOrder; ++k)
102 for (label i = 0; i <= pOrder - k; ++i)
103 for (label j = 0; j <= pOrder - k; ++j)
104 ++count;
105
106 return count;
107 }
108
109 // --------------------------------------------------
110 // Tetrahedron
111 // Constraint: i + j + k <= r
112 // --------------------------------------------------
113 case dgCellType::TET:
114 {
115 for (label i = 0; i <= pOrder; ++i)
116 for (label j = 0; j <= pOrder - i; ++j)
117 for (label k = 0; k <= pOrder - i - j; ++k)
118 ++count;
119
120 return count;
121 }
122
123 case dgCellType::INVALID:
124 default:
125 {
126 FatalErrorInFunction
127 << "Unsupported or invalid cell type for basis function: "
128 << type << nl
129 << exit(FatalError);
130
131 return -1; // suppress compiler warning
132 }
133 }
134}

187inline void computeHexBasisAndDerivatives
188(
189 const scalar x,
190 const scalar y,
191 const scalar z,
192 const label pOrder,
193 List<scalar>& basis,
194 List<scalar>& dBasis_dx,
195 List<scalar>& dBasis_dy,
196 List<scalar>& dBasis_dz
197)
198{
199 // --------------------------------------------------
200 // Orthogonal Legendre basis on reference hexahedron
201 // Basis definition:
202 // psi_{i,j,k} = P_i(x) P_j(y) P_k(z)
203 // with i + j + k <= pOrder
204 // --------------------------------------------------
205
206 using boost::math::jacobi;
207 using boost::math::jacobi_prime;
208
209 // Number of basis functions for total-degree HEX
210 const label nBasis = getNumBasis(pOrder, dgCellType::HEX);
211
212 basis.setSize(nBasis);
213 dBasis_dx.setSize(nBasis);
214 dBasis_dy.setSize(nBasis);
215 dBasis_dz.setSize(nBasis);
216
217 label idx = 0;
218
219 // Loop over total degree
220 for (label i = 0; i <= pOrder; ++i)
221 {
222 const scalar Pi = jacobi(static_cast<unsigned>(i), 0.0, 0.0, x);
223 const scalar dPi = jacobi_prime(static_cast<unsigned>(i), 0.0, 0.0, x);
224
225 for (label j = 0; j <= pOrder - i; ++j)
226 {
227 const scalar Pj = jacobi(static_cast<unsigned>(j), 0.0, 0.0, y);
228 const scalar dPj = jacobi_prime(static_cast<unsigned>(j), 0.0, 0.0, y);
229
230 for (label k = 0; k <= pOrder - i - j; ++k)
231 {
232 const scalar Pk = jacobi(static_cast<unsigned>(k), 0.0, 0.0, z);
233 const scalar dPk = jacobi_prime(static_cast<unsigned>(k), 0.0, 0.0, z);
234
235 // Basis value
236 basis[idx] = Pi * Pj * Pk;
237
238 // Derivatives w.r.t reference coordinates
239 dBasis_dx[idx] = dPi * Pj * Pk;
240 dBasis_dy[idx] = Pi * dPj * Pk;
241 dBasis_dz[idx] = Pi * Pj * dPk;
242
243 ++idx;
244 }
245 }
246 }
247}

306inline void computePrismBasisAndDerivatives
307(
308 const scalar x,
309 const scalar y,
310 const scalar z,
311 const label pOrder,
312 List<scalar>& basis,
313 List<scalar>& dBasis_dx,
314 List<scalar>& dBasis_dy,
315 List<scalar>& dBasis_dz
316)
317{
318 using boost::math::jacobi;
319 using boost::math::jacobi_prime;
320
321 const label nBasis = getNumBasis(pOrder, dgCellType::PRISM);
322
323 basis.setSize(nBasis);
324 dBasis_dx.setSize(nBasis);
325 dBasis_dy.setSize(nBasis);
326 dBasis_dz.setSize(nBasis);
327
328 label idx = 0;
329
330 // Precompute common terms
331 const scalar oneMinusY = 1.0 - y;
332 const scalar invOneMinusY = 1.0 / oneMinusY;
333
334 // Mapped coordinates
335 const scalar xi = 2.0 * x * invOneMinusY - 1.0;
336 const scalar eta = 2.0 * y - 1.0;
337 const scalar zeta = 2.0 * z - 1.0;
338
339 for (label i = 0; i <= pOrder; ++i)
340 {
341 // P_i^{0,0}( (2x)/(1-y) - 1 )
342 const scalar Pi = jacobi(static_cast<unsigned>(i), 0.0, 0.0, xi);
343 const scalar dPi = jacobi_prime(static_cast<unsigned>(i), 0.0, 0.0, xi);
344
345 // (1 - y)^i and derivative
346 const scalar oneMinusYPowI =
347 (i == 0 ? 1.0 : std::pow(oneMinusY, i));
348
349 const scalar dOneMinusYPowI_dy =
350 (i == 0 ? 0.0 : -i * std::pow(oneMinusY, i - 1));
351
352 for (label j = 0; j <= pOrder - i; ++j)
353 {
354 // P_j^{2i+1,0}(2y - 1)
355 const scalar alpha = 2.0 * i + 1.0;
356
357 const scalar Pj =
358 jacobi(static_cast<unsigned>(j), alpha, 0.0, eta);
359
360 const scalar dPj =
361 jacobi_prime(static_cast<unsigned>(j), alpha, 0.0, eta);
362
363 for (label k = 0; k <= pOrder - i - j; ++k)
364 {
365 // P_k^{0,0}(2z - 1)
366 const scalar Pk =
367 jacobi(static_cast<unsigned>(k), 0.0, 0.0, zeta);
368
369 const scalar dPk =
370 jacobi_prime(static_cast<unsigned>(k), 0.0, 0.0, zeta);
371
372 // --------------------------------------------------
373 // Basis value
374 // --------------------------------------------------
375 basis[idx] =
376 Pi * oneMinusYPowI * Pj * Pk;
377
378 // --------------------------------------------------
379 // ∂/∂x
380 // xi = 2x / (1-y) - 1
381 // --------------------------------------------------
382 dBasis_dx[idx] =
383 dPi * (2.0 * invOneMinusY)
384 * oneMinusYPowI * Pj * Pk;
385
386 // --------------------------------------------------
387 // ∂/∂y
388 // --------------------------------------------------
389 const scalar dxi_dy =
390 2.0 * x * invOneMinusY * invOneMinusY;
391
392 dBasis_dy[idx] =
393 // derivative of Pi(xi)
394 dPi * dxi_dy * oneMinusYPowI * Pj * Pk
395 + // derivative of (1-y)^i
396 Pi * dOneMinusYPowI_dy * Pj * Pk
397 + // derivative of Pj(eta)
398 Pi * oneMinusYPowI * dPj * 2.0 * Pk;
399
400 // --------------------------------------------------
401 // ∂/∂z
402 // zeta = 2z - 1
403 // --------------------------------------------------
404 dBasis_dz[idx] =
405 Pi * oneMinusYPowI * Pj * dPk * 2.0;
406
407 ++idx;
408 }
409 }
410 }

476inline void computePyramidBasisAndDerivatives
477(
478 const scalar x,
479 const scalar y,
480 const scalar z,
481 const label pOrder,
482 List<scalar>& basis,
483 List<scalar>& dBasis_dx,
484 List<scalar>& dBasis_dy,
485 List<scalar>& dBasis_dz
486)
487{
488 using boost::math::jacobi;
489 using boost::math::jacobi_prime;
490
491 const label nBasis = getNumBasis(pOrder, dgCellType::PYRAMID);
492
493 basis.setSize(nBasis);
494 dBasis_dx.setSize(nBasis);
495 dBasis_dy.setSize(nBasis);
496 dBasis_dz.setSize(nBasis);
497
498 label idx = 0;
499
500 const scalar oneMinusZ = 1.0 - z;
501 const scalar invOneMinusZ = 1.0 / oneMinusZ;
502
503 // Mapped coordinates
504 const scalar xi = x * invOneMinusZ;
505 const scalar eta = y * invOneMinusZ;
506 const scalar zeta = 2.0 * z - 1.0;
507
508 for (label i = 0; i <= pOrder; ++i)
509 {
510 const scalar Pi = jacobi(static_cast<unsigned>(i), 0.0, 0.0, xi);
511 const scalar dPi = jacobi_prime(static_cast<unsigned>(i), 0.0, 0.0, xi);
512
513 for (label j = 0; j <= pOrder; ++j)
514 {
515 const scalar Pj = jacobi(static_cast<unsigned>(j), 0.0, 0.0, eta);
516 const scalar dPj = jacobi_prime(static_cast<unsigned>(j), 0.0, 0.0, eta);
517
518 const label m = max(i, j);
519
520 // (1 - z)^{max(i,j)}
521 const scalar powTerm =
522 (m == 0 ? 1.0 : std::pow(oneMinusZ, m));
523
524 const scalar dPowTerm_dz =
525 (m == 0 ? 0.0 : -m * std::pow(oneMinusZ, m - 1));
526
527 for (label k = 0; k <= pOrder - m; ++k)
528 {
529 const scalar alpha = 2.0 * m + 2.0;
530
531 const scalar Pk =
532 jacobi(static_cast<unsigned>(k), alpha, 0.0, zeta);
533
534 const scalar dPk =
535 jacobi_prime(static_cast<unsigned>(k), alpha, 0.0, zeta);
536
537 // --------------------------------------------------
538 // Basis value
539 // --------------------------------------------------
540 basis[idx] = Pi * Pj * powTerm * Pk;
541
542 // --------------------------------------------------
543 // ∂/∂x
544 // xi = x / (1 - z)
545 // --------------------------------------------------
546 dBasis_dx[idx] =
547 dPi * invOneMinusZ * Pj * powTerm * Pk;
548
549 // --------------------------------------------------
550 // ∂/∂y
551 // eta = y / (1 - z)
552 // --------------------------------------------------
553 dBasis_dy[idx] =
554 Pi * dPj * invOneMinusZ * powTerm * Pk;
555
556 // --------------------------------------------------
557 // ∂/∂z
558 // --------------------------------------------------
559 const scalar dxi_dz = x * invOneMinusZ * invOneMinusZ;
560 const scalar deta_dz = y * invOneMinusZ * invOneMinusZ;
561
562 dBasis_dz[idx] =
563 // derivative through xi
564 dPi * dxi_dz * Pj * powTerm * Pk
565 + // derivative through eta
566 Pi * dPj * deta_dz * powTerm * Pk
567 + // derivative of (1 - z)^m
568 Pi * Pj * dPowTerm_dz * Pk
569 + // derivative of Pk(2z - 1)
570 Pi * Pj * powTerm * dPk * 2.0;
571
572 ++idx;
573 }
574 }
575 }

636inline void computeTetBasisAndDerivatives
637(
638 const scalar x,
639 const scalar y,
640 const scalar z,
641 const label pOrder,
642 List<scalar>& basis,
643 List<scalar>& dBasis_dx,
644 List<scalar>& dBasis_dy,
645 List<scalar>& dBasis_dz
646)
647{
648 using boost::math::jacobi;
649 using boost::math::jacobi_prime;
650
651 const label nBasis = getNumBasis(pOrder, dgCellType::TET);
652
653 basis.setSize(nBasis);
654 dBasis_dx.setSize(nBasis);
655 dBasis_dy.setSize(nBasis);
656 dBasis_dz.setSize(nBasis);
657
658 label idx = 0;
659
660 const scalar oneMinusZ = 1.0 - z;
661 const scalar oneMinusYZ = 1.0 - y - z;
662 const scalar invOneMinusZ = 1.0 / oneMinusZ;
663 const scalar invOneMinusYZ = 1.0 / oneMinusYZ;
664
665 // Mapped coordinates
666 const scalar xi = 2.0 * x * invOneMinusYZ - 1.0;
667 const scalar eta = 2.0 * y * invOneMinusZ - 1.0;
668 const scalar zeta = 2.0 * z - 1.0;
669
670 for (label i = 0; i <= pOrder; ++i)
671 {
672 const scalar Pi = jacobi(static_cast<unsigned>(i), 0.0, 0.0, xi);
673 const scalar dPi = jacobi_prime(static_cast<unsigned>(i), 0.0, 0.0, xi);
674
675 const scalar oneMinusYZPowI =
676 (i == 0 ? 1.0 : std::pow(oneMinusYZ, i));
677
678 const scalar dOneMinusYZPowI_dy =
679 (i == 0 ? 0.0 : -i * std::pow(oneMinusYZ, i - 1));
680
681 const scalar dOneMinusYZPowI_dz =
682 dOneMinusYZPowI_dy; // same derivative w.r.t y and z
683
684 for (label j = 0; j <= pOrder - i; ++j)
685 {
686 const scalar alpha_j = 2.0 * i + 1.0;
687
688 const scalar Pj =
689 jacobi(static_cast<unsigned>(j), alpha_j, 0.0, eta);
690
691 const scalar dPj =
692 jacobi_prime(static_cast<unsigned>(j), alpha_j, 0.0, eta);
693
694 const scalar oneMinusZPowJ =
695 (j == 0 ? 1.0 : std::pow(oneMinusZ, j));
696
697 const scalar dOneMinusZPowJ_dz =
698 (j == 0 ? 0.0 : -j * std::pow(oneMinusZ, j - 1));
699
700 for (label k = 0; k <= pOrder - i - j; ++k)
701 {
702 const scalar alpha_k = 2.0 * (i + j) + 2.0;
703
704 const scalar Pk =
705 jacobi(static_cast<unsigned>(k), alpha_k, 0.0, zeta);
706
707 const scalar dPk =
708 jacobi_prime(static_cast<unsigned>(k), alpha_k, 0.0, zeta);
709
710 // --------------------------------------------------
711 // Basis value
712 // --------------------------------------------------
713 basis[idx] =
714 Pi * oneMinusYZPowI *
715 Pj * oneMinusZPowJ *
716 Pk;
717
718 // --------------------------------------------------
719 // ∂/∂x
720 // xi = 2x / (1 - y - z) - 1
721 // --------------------------------------------------
722 dBasis_dx[idx] =
723 dPi * (2.0 * invOneMinusYZ) *
724 oneMinusYZPowI * Pj * oneMinusZPowJ * Pk;
725
726 // --------------------------------------------------
727 // ∂/∂y
728 // --------------------------------------------------
729 const scalar dxi_dy =
730 2.0 * x * invOneMinusYZ * invOneMinusYZ;
731
732 dBasis_dy[idx] =
733 // through xi
734 dPi * dxi_dy * oneMinusYZPowI * Pj * oneMinusZPowJ * Pk
735 + // derivative of (1 - y - z)^i
736 Pi * dOneMinusYZPowI_dy * Pj * oneMinusZPowJ * Pk
737 + // through eta
738 Pi * oneMinusYZPowI *
739 dPj * (2.0 * invOneMinusZ) *
740 oneMinusZPowJ * Pk;
741
742 // --------------------------------------------------
743 // ∂/∂z
744 // --------------------------------------------------
745 const scalar dxi_dz =
746 2.0 * x * invOneMinusYZ * invOneMinusYZ;
747
748 const scalar deta_dz =
749 2.0 * y * invOneMinusZ * invOneMinusZ;
750
751 dBasis_dz[idx] =
752 // through xi
753 dPi * dxi_dz * oneMinusYZPowI * Pj * oneMinusZPowJ * Pk
754 + // derivative of (1 - y - z)^i
755 Pi * dOneMinusYZPowI_dz * Pj * oneMinusZPowJ * Pk
756 + // through eta
757 Pi * oneMinusYZPowI *
758 dPj * deta_dz *
759 oneMinusZPowJ * Pk
760 + // derivative of (1 - z)^j
761 Pi * oneMinusYZPowI * Pj * dOneMinusZPowJ_dz * Pk
762 + // through zeta
763 Pi * oneMinusYZPowI * Pj * oneMinusZPowJ *
764 dPk * 2.0;
765
766 ++idx;
767 }
768 }
769 }
