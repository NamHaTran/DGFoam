\DoxyCodeLine{00031\ \textcolor{comment}{\ \ \ \ HLLC\ (Harten–Lax–van\ Leer–Contact)\ approximate\ Riemann\ solver.}}
\DoxyCodeLine{00032\ \textcolor{comment}{}}
\DoxyCodeLine{00033\ \textcolor{comment}{\ \ \ \ This\ flux\ restores\ the\ missing\ contact\ and\ shear\ waves\ that\ are\ removed\ in}}
\DoxyCodeLine{00034\ \textcolor{comment}{\ \ \ \ the\ HLLE\ formulation.\ HLLC\ introduces\ a\ middle\ wave\ S*\ and\ two}}
\DoxyCodeLine{00035\ \textcolor{comment}{\ \ \ \ reconstructed\ intermediate\ states\ (U*\_L\ and\ U*\_R),\ allowing\ accurate}}
\DoxyCodeLine{00036\ \textcolor{comment}{\ \ \ \ resolution\ of\ contact\ discontinuities\ and\ tangential\ shear\ layers.}}
\DoxyCodeLine{00037\ \textcolor{comment}{}}
\DoxyCodeLine{00038\ \textcolor{comment}{\ \ \ \ Compared\ to\ HLLE:}}
\DoxyCodeLine{00039\ \textcolor{comment}{\ \ \ \ -\/\ HLLC\ is\ significantly\ less\ diffusive.}}
\DoxyCodeLine{00040\ \textcolor{comment}{\ \ \ \ -\/\ Captures\ contact\ discontinuities\ sharply.}}
\DoxyCodeLine{00041\ \textcolor{comment}{\ \ \ \ -\/\ Preserves\ tangential\ velocity\ continuity\ (important\ for\ momentum\ flux).}}
\DoxyCodeLine{00042\ \textcolor{comment}{\ \ \ \ -\/\ Requires\ computing\ a\ star\ region\ and\ wave\ strengths\ C\_L,\ C\_R.}}
\DoxyCodeLine{00043\ \textcolor{comment}{}}
\DoxyCodeLine{00044\ \textcolor{comment}{\ \ \ \ Procedure\ (local\ 1D\ Riemann\ problem\ along\ face\ normal\ n):}}
\DoxyCodeLine{00045\ \textcolor{comment}{\ \ \ \ 1.\ Project\ left\ and\ right\ physical\ fluxes\ onto\ n.}}
\DoxyCodeLine{00046\ \textcolor{comment}{\ \ \ \ 2.\ Compute\ left/right\ normal\ velocities\ and\ sound\ speeds.}}
\DoxyCodeLine{00047\ \textcolor{comment}{\ \ \ \ 3.\ Estimate\ wave\ speeds\ S\_L\ and\ S\_R\ using\ either:}}
\DoxyCodeLine{00048\ \textcolor{comment}{\ \ \ \ \ \ \ \ \ *\ Davis\ side\ speeds,\ or}}
\DoxyCodeLine{00049\ \textcolor{comment}{\ \ \ \ \ \ \ \ \ *\ Roe–Einfeldt\ averages\ (U\_roe,\ H\_roe,\ a\_roe).}}
\DoxyCodeLine{00050\ \textcolor{comment}{\ \ \ \ 4.\ Compute\ the\ contact\ wave\ speed\ S*\ based\ on\ pressure\ and\ velocity\ jumps.}}
\DoxyCodeLine{00051\ \textcolor{comment}{\ \ \ \ 5.\ Compute\ the\ scaling\ coefficients:}}
\DoxyCodeLine{00052\ \textcolor{comment}{\ \ \ \ \ \ \ \ \ C\_L\ =\ rho\_L\ (S\_L\ -\/\ Un\_L)\ /\ (S\_L\ -\/\ S*)}}
\DoxyCodeLine{00053\ \textcolor{comment}{\ \ \ \ \ \ \ \ \ C\_R\ =\ rho\_R\ (S\_R\ -\/\ Un\_R)\ /\ (S\_R\ -\/\ S*)}}
\DoxyCodeLine{00054\ \textcolor{comment}{\ \ \ \ 6.\ Construct\ the\ intermediate\ star\ states\ U*\_L\ and\ U*\_R:}}
\DoxyCodeLine{00055\ \textcolor{comment}{\ \ \ \ \ \ \ \ \ -\/\ For\ mass:\ \ \ \ \ U*\ =\ C}}
\DoxyCodeLine{00056\ \textcolor{comment}{\ \ \ \ \ \ \ \ \ -\/\ For\ momentum:\ U*\ =\ C\ (\ S*\ n\ +\ U\_t\ ),\ preserving\ tangential\ velocity}}
\DoxyCodeLine{00057\ \textcolor{comment}{\ \ \ \ \ \ \ \ \ -\/\ For\ energy:\ \ \ U*\ =\ C\ (\ e\ +\ corrective\ term\ )}}
\DoxyCodeLine{00058\ \textcolor{comment}{\ \ \ \ \ \ \ \ \ -\/\ For\ scalars:\ \ U*\ =\ C\ q}}
\DoxyCodeLine{00059\ \textcolor{comment}{\ \ \ \ 7.\ Construct\ the\ flux\ in\ each\ region:}}
\DoxyCodeLine{00060\ \textcolor{comment}{\ \ \ \ \ \ \ \ \ f\ =\ f\_L,\ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ if\ S\_L\ >=\ 0}}
\DoxyCodeLine{00061\ \textcolor{comment}{\ \ \ \ \ \ \ \ \ f\ =\ f*\_L,\ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ if\ S\_L\ <=\ 0\ <=\ S*}}

\DoxyCodeLine{00064\ \textcolor{comment}{\ \ \ \ \ \ \ where\ f*\_L\ and\ f*\_R\ are\ star-\/region\ fluxes.}}
